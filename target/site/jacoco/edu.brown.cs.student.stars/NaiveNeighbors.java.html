package edu.brown.cs.student.stars;

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

/**
 * Class for the naive_neighbors command.
 */

public class NaiveNeighbors implements Command {

  @Override
  public void execute(String params) {

    if (Common.getAllStars() == null) {
      System.err.println("ERROR: Star data not loaded yet");
    } else {

      if (Common.checkRegex(params,
          "^\\d+\\s-?\\d+(?:\\.\\d+)?\\s-?\\d+(?:\\.\\d+)?\\s-?\\d+(?:\\.\\d+)?")) {

        String[] p = params.split(" ");
        List<Star> stars = findKNearestNeighbor(
            Integer.parseInt(p[0]),
            Double.parseDouble(p[1]),
            Double.parseDouble(p[2]),
            Double.parseDouble(p[3]));

        String output = Star.toString(stars);
        if (!output.equals("")) {
          System.out.println(Star.toString(stars));
        }

      } else if (Common.checkRegex(params, "^\\d+\\s\"([^\"])*\"")) {

        String[] p = params.split(" ", 2);
        List<Star> stars = findKNearestNeighbor(Integer.parseInt(p[0]), Common.removeQuotes(p[1]));

        String output = Star.toString(stars);
        if (!output.equals("")) {
          System.out.println(Star.toString(stars));
        }

      } else {
        System.err.println("ERROR: Wrong parameters");
      }
    }
  }

  /**
   * Finds the closest k stars to a given location.
   *
   * @param k Number of neighbors to be found
   * @param x x coordinates of the location
   * @param y y coordinates of the location
   * @param z z coordinates of the location
   * @return A list of k closest stars
   */
  public List<Star> findKNearestNeighbor(int k, double x, double y, double z) {
    double[] currentCoordinate = new double[]{x, y, z};
    Common.setStarDistanceFrom(currentCoordinate, Common.getAllStars());
    Collections.sort(Common.getAllStars());
    return getKStars(k, Common.getAllStars());
  }

  /**
   * Finds the closest k stars to a given star.
   *
   * @param k    Number of neighbors to be found
   * @param name Name of a star
   * @return A list of k closest stars
   */
  public List<Star> findKNearestNeighbor(int k, String name) {
    Star currentStar = Common.findStarByName(name);
    List<Star> starsCopy = new ArrayList<>();
    if (currentStar != null) {
      starsCopy = Common.copyStarListExcept(name);
      Common.setStarDistanceFrom(currentStar.getCoordinate(), starsCopy);
      Collections.sort(starsCopy);
    }
    return getKStars(k, starsCopy);
  }

  /**
   * Extracts the first k stars from a list of stars sorted according to distance.
   * Randomly choose one to be returned if there is a tie in distance.
   *
   * @param k     Number of stars to be returned
   * @param stars A list of stars that have been sorted in ascending order of distance
   * @return First k stars
   */
  public List<Star> getKStars(int k, List<Star> stars) {
    List<Star> output = new ArrayList<>();

    if (stars.size() != 0) {
      List<Star> neighbors = new ArrayList<>();

      int i = 0;
      Star tempStar;
      double maxDistanceSoFar = 0;

      // take out the ones that potentially need to be printed
      while ((i < stars.size())
          && ((stars.get(i).getDistance() <= maxDistanceSoFar) || (neighbors.size() < k))) {
        tempStar = stars.get(i);
        neighbors.add(tempStar);
        maxDistanceSoFar = tempStar.getDistance();
        i++;
      }

      // take out the untied stars first
      while (neighbors.size() != 0) {
        boolean a = (neighbors.get(0).getDistance() < maxDistanceSoFar);
        if (neighbors.get(0).getDistance() < maxDistanceSoFar) {
          output.add(neighbors.get(0));
          neighbors.remove(neighbors.get(0));
          k--;
        } else {
          break;
        }
      }

      // for the tied ones, shuffle before taking out
      i = 0;
      while ((neighbors.size() != 0) && (i < k)) {
        Collections.shuffle(neighbors);
        output.add(neighbors.get(0));
        neighbors.remove(neighbors.get(0));
        i++;
      }
    }

    return output;

  }

}
