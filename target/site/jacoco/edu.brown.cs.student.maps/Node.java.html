package edu.brown.cs.student.maps;

import edu.brown.cs.student.stars.HasCoordinate;

import java.sql.ResultSet;
import java.sql.SQLException;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;
import java.util.Objects;

/**
 * Class that represents a graph node.
 */
public class Node implements Vertex, HasCoordinate, Comparable<Node> {

  private String id;
  private double[] coordinate;
  private double distance;

  /**
   * Constructs a node.
   * @param id node id
   * @param latitude node's latitude
   * @param longitude node's longitude
   */
  public Node(String id, double latitude, double longitude) {
    this.id = id;
    this.coordinate = new double[]{latitude, longitude};
    this.distance = -1;
  }

  @Override
  public double[] getCoordinate() {
    return Arrays.copyOf(coordinate, coordinate.length);
  }

  @Override
  public double getDistance() {
    return distance;
  }

  @Override
  public void setDistance(double distance) {
    this.distance = distance;
  }

  /**
   * Finds all neighbors of the current node.
   * @return a list of paths from the current node to its neighbors
   * @throws SQLException if any errors occur during SQL query
   */
  @Override
  public List<Path> findNeighbors() throws SQLException {
    List<Path> paths = new ArrayList<>();
    List<String> vals = new ArrayList<>();
    vals.add(this.id);
    ResultSet rs = MapCommand.getDatabase().runSQLQuery(
            "SELECT way.id, way.end, node.latitude, node.longitude "
                + "FROM way JOIN node ON "
                + "(way.start = \"" + this.id + "\") and (node.id = way.end);");
    while (rs.next()) {
      String neighborID = rs.getString("end");
      String wayID = rs.getString("id");
      Double neighborLat = rs.getDouble("latitude");
      Double neighborLon = rs.getDouble("longitude");
      Double neighborDist = Haversine.distance(this.coordinate[0],
              this.coordinate[1], neighborLat, neighborLon);
      Node neighbor = new Node(neighborID, neighborLat, neighborLon);
      Path pathToNeighbor = new Path(new Node(this.id, this.coordinate[0], this.coordinate[1]),
              neighbor, neighborDist, wayID);
      paths.add(pathToNeighbor);
    }
    return paths;
  }

  @Override
  public Path edgeToSelf(double weight) {
    return new Path(this, this, weight, "");
  }

  /**
   * Overrides hashCode.
   * @return Positive number if greater, 0 if same, negative if less
   */
  @Override
  public int hashCode() {
    return Objects.hash(getSigFields());
  }

  /**
   * Overriding equality for Node object.
   *
   * @param secondNode Node to be checked if equal to this Node.
   * @return True if equal, false if unequal.
   */
  @Override
  public boolean equals(Object secondNode) {
    if (this == secondNode) {
      return true;
    }
    if (!(secondNode instanceof Node)) {
      return false;
    }
    Node second = (Node) secondNode;
    for (int i = 0; i < this.getSigFields().length; i++) {
      if (!Objects.equals(this.getSigFields()[i], second.getSigFields()[i])) {
        return false;
      }
    }
    return true;
  }

  /**
   * Gets the significant fields as an array of objects.
   * @return significant field of node object
   */
  private Object[] getSigFields() {
    Object[] result = {id, distance};
    return result;
  }

  /**
   * Compares two stars according to their distance value.
   * @param n Another node
   * @return 0 if distance is equal, 1 if larger, -1 if smaller
   */
  @Override
  public int compareTo(Node n) {
    return Double.compare(distance, n.getDistance());
  }

  @Override
  public String getID() {
    return id;
  }

  @Override
  public String toString() {
    return id;
  }
}
