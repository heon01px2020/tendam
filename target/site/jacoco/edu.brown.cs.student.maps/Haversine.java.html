package edu.brown.cs.student.maps;

/**
 * Class for calculating haversine distance.
 */
public final class Haversine {
  private static final int RADIUS_EARTH = 6371; // Approximation of Earth radius in km

  private Haversine() {
  }

  /**
   * Calculates the haversine distance between two coordinates.
   * @param lat1 latitude of point 1
   * @param lon1 longitude of point 1
   * @param lat2 latitude of point 2
   * @param lon2 longitude of point 2
   * @return haversine distance
   */
  public static double distance(double lat1, double lon1, double lat2, double lon2) {

    double lat1rad = Math.toRadians(lat1);
    double lat2rad = Math.toRadians(lat2);

    double dLat  = Math.toRadians((lat2 - lat1));
    double dLon = Math.toRadians((lon2 - lon1));

    // h = haversine of central angle
    double h = haversine(dLat) + Math.cos(lat1rad) * Math.cos(lat2rad) * haversine(dLon);
    double d = 2 * RADIUS_EARTH * Math.asin(Math.sqrt(h));

    return d;
  }

  /**
   * Calculates the haversine based on formula.
   * @param v a value
   * @return haversine
   */
  public static double haversine(double v) {
    return Math.pow(Math.sin(v / 2), 2);
  }
}
