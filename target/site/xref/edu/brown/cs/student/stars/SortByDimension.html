3   import java.io.Serializable;
4   import java.util.Comparator;
5   
6   /**
7    * Class that can be sorted by dimension.
8    * @param <T> Object type that extends CoordinateGettable and implements comparator and serializable
9    */
10  public class SortByDimension<T extends HasCoordinate> implements Comparator<T>, Serializable {
11    private int dimension;
12    //private int totalDimension;
13    private static final long serialVersionUID = 1234L;
14  
15  //  /**
16  //   * Creates a new SortByDimension object.
17  //   * @param dimension Integer representing current dimension
18  //   * @param totalDimension Integer representing the total dimension
19  //   */
20    public SortByDimension(int dimension) {
21      this.dimension = dimension;
22      //this.totalDimension = totalDimension;
23    }
24  
25    /**
26     * Compares two objects based on their dimension.
27     * @param obj1 First object
28     * @param obj2 Second object
29     * @return Positive number of first object is greater, 0 if same, negative if smaller
30     */
31    public int compare(T obj1, T obj2) {
32      return Double.compare(
33              obj1.getCoordinate()[this.dimension],
34              obj2.getCoordinate()[this.dimension]);
35    }
36  }
