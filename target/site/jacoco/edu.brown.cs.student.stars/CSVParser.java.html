package edu.brown.cs.student.stars;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.Reader;
import java.util.ArrayList;
import java.util.LinkedHashMap;
import java.util.List;

/**
 * Class to read and parse CSV files.
 */
public class CSVParser {
  private BufferedReader br;
  private boolean hasNext = true;

  /**
   * Constructor to set up a CSV parser.
   *
   * @param reader a Reader object
   */
  public CSVParser(Reader reader) {
    this.br = new BufferedReader(reader);
  }

  /**
   * Reads and parses an entire CSV file.
   *
   * @return A list of LinkedHashMap containing parsed data,
   * where each HashMap represents a row in the CSV
   * @throws IOException if accidents happen during read
   */
  public List<LinkedHashMap<String, String>> parseAll() throws IOException {
    String[] columnHeaders = parseLine(readNextLine());
    List<LinkedHashMap<String, String>> allElements = new ArrayList<>();
    while (hasNext) {
      String nextLine = readNextLine();
      LinkedHashMap<String, String> newElement = new LinkedHashMap<>();
      if (nextLine != null) {
        String[] nextLineParsed = parseLine(nextLine);
        for (int i = 0; i < columnHeaders.length; i++) {
          newElement.put(columnHeaders[i], nextLineParsed[i]);
        }
        allElements.add(newElement);
      }
    }
    return allElements;
  }

  /**
   * Parses a line by splitting the columns.
   *
   * @param line A line read from a CSV file
   * @return Items from the line stored in an array
   */
  public static String[] parseLine(String line) {
    return (line == null) ? null : line.split(",", -1);
  }

  /**
   * Reads the next line from a file.
   *
   * @return The next line from the file, or null if bottom of file is reached
   * @throws IOException if accidents happen during read
   */
  private String readNextLine() throws IOException {
    String nextLine = br.readLine();
    if (nextLine == null) {
      hasNext = false;
    }
    return hasNext ? nextLine : null;
  }

  /**
   * Checks the format of data fields parsed from a CSV file
   * against the manually inputted expectedHeaders and expectedDataFormat.
   *
   * @param data               Data parsed from a CSV file
   * @param expectedHeaders    List of String containing the expected column headers from a CSV file
   * @param expectedDataFormat List of String containing the expected data format of each field,
   *                           represented in Regex
   * @return a boolean indicating if all data fields match the expected formats
   */
  public static boolean checkFormat(List<LinkedHashMap<String, String>> data,
                                    List<String> expectedHeaders,
                                    List<String> expectedDataFormat) {
    if (data.size() != 0) {
      List<String> keys = new ArrayList<>(data.get(0).keySet());
      if (!keys.equals(expectedHeaders)) {
        return false;
      } else {
        for (int i = 0; i < data.size(); i++) {
          List<String> values = new ArrayList<>();
          for (String key : keys) {
            values.add(data.get(i).get(key));
          }
          if (!Common.checkRegex(values, expectedDataFormat)) {
            return false;
          }
        }
      }
    }
    return true;
  }

}
