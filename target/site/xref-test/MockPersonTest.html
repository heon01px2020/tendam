19    @Test
20    public void testToStringList() {
21      MockPerson m1 = new MockPerson("Jenny", "Yu", "2003/02/12",
22          "jenny_yu2@brown.edu", "Female", "Somewhere");
23      MockPerson m2 = new MockPerson("Bob", "Kim", "2003/02/12",
24          "jenny_yu2@brown.edu", "Female", "Somewhere");
25      List<MockPerson> mockPeople = new ArrayList<>();
26      mockPeople.add(m1);
27      mockPeople.add(m2);
28  
29      assertEquals(MockPerson.toString(mockPeople),
30          "Name: Jenny Yu; Datetime: 2003/02/12; Email Address: jenny_yu2@brown.edu; " +
31          "Gender: Female; Street Address: Somewhere\n" +
32              "Name: Bob Kim; Datetime: 2003/02/12; Email Address: jenny_yu2@brown.edu; " +
33              "Gender: Female; Street Address: Somewhere");
34    }
