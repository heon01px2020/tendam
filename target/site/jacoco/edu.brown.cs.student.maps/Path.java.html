package edu.brown.cs.student.maps;

import java.util.Objects;

/**
 * Class that represents a map path.
 * @param <T> any type that implements the vertex interface.
 */
public class Path<T extends Vertex> implements Comparable<Path>, Edge {
  private T start;
  private T end;
  private double weightedDistance;
  private String id;

  /**
   * Constructs a path.
   * @param start starting vertex
   * @param end end vertex
   * @param weightedDistance weighted distance of the path
   * @param id path id
   */
  public Path(T start, T end, double weightedDistance, String id) {
    this.start = start;
    this.end = end;
    this.weightedDistance = weightedDistance;
    this.id = id;
  }

  @Override
  public T getStart() {
    return start;
  }

  @Override
  public T getEnd() {
    return end;
  }

  @Override
  public double getWeightedDistance() {
    return weightedDistance;
  }

  @Override
  public void setWeightedDistance(double weightedDistance) {
    this.weightedDistance = weightedDistance;
  }

  @Override
  public String getID() {
    return id;
  }

  /**
   * Overrides hashCode.
   * @return Positive number if greater, 0 if same, negative if less
   */
  @Override
  public int hashCode() {
    return Objects.hash(getSigFields());
  }

  /**
   * Overriding equality for Path object.
   * @param secondPath Path to be checked if equal to this Path.
   * @return True if equal, false if unequal.
   */
  @Override
  public boolean equals(Object secondPath) {
    if (this == secondPath) {
      return true;
    }
    if (!(secondPath instanceof Path)) {
      return false;
    }
    Path second = (Path) secondPath;
    for (int i = 0; i < this.getSigFields().length; ++i) {
      if (!Objects.equals(this.getSigFields()[i], second.getSigFields()[i])) {
        return false;
      }
    }
    return true;
  }

  /**
   * Gets the significant fields as an array of objects.
   * @return significant field of Path object
   */
  private Object[] getSigFields() {
    Object[] result = {start, end, weightedDistance, id};
    return result;
  }

  @Override
  public String toString() {
    return start + " -> " + end + " : " + id;
  }

  @Override
  public int compareTo(Path o) {
    return Double.compare(weightedDistance, o.getWeightedDistance());
  }
}
