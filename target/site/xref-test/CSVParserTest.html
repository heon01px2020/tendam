46    @Test
47    public void testCheckFormat() {
48      List<String> headers = new ArrayList<>(Arrays.asList("id", "name"));
49      List<String> format = new ArrayList<>(Arrays.asList("^\\d+", "(?:.*)?"));
50  
51      LinkedHashMap correctRow = new LinkedHashMap();
52      correctRow.put("id", "2");
53      correctRow.put("name", "hi");
54  
55      LinkedHashMap extraElementRow = new LinkedHashMap();
56      extraElementRow.put("id", "2");
57      extraElementRow.put("name", "hi");
58      extraElementRow.put("distance", "5");
59  
60      LinkedHashMap wrongHeaderRow = new LinkedHashMap();
61      wrongHeaderRow.put("id", "2");
62      wrongHeaderRow.put("NAME", "hi");
63  
64      LinkedHashMap wrongFormatRow = new LinkedHashMap();
65      wrongHeaderRow.put("id", "xxx");
66      wrongHeaderRow.put("name", "h");
67  
68      List<LinkedHashMap<String, String>> correct1 = new ArrayList<>();
69      correct1.add(correctRow);
70      List<LinkedHashMap<String, String>> emptyData = new ArrayList<>();
71      List<LinkedHashMap<String, String>> wrongHeader = new ArrayList<>();
72      wrongHeader.add(wrongHeaderRow);
73      List<LinkedHashMap<String, String>> extraElement = new ArrayList<>();
74      extraElement.add(extraElementRow);
75      List<LinkedHashMap<String, String>> wrongFormat = new ArrayList<>();
76      wrongFormat.add(wrongFormatRow);
77  
78      assertTrue(CSVParser.checkFormat(correct1, headers, format));
79      assertTrue(CSVParser.checkFormat(emptyData, headers, format));
80      assertFalse(CSVParser.checkFormat(wrongHeader, headers, format));
81      assertFalse(CSVParser.checkFormat(extraElement, headers, format));
82      assertFalse(CSVParser.checkFormat(wrongFormat, headers, format));
