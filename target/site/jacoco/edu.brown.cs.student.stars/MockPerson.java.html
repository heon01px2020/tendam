package edu.brown.cs.student.stars;

import java.util.List;

/**
 * Class that represents a person with designated attributes.
 */
public class MockPerson {
  private String firstName;
  private String lastName;
  private String datetime;
  private String emailAddress;
  private String gender;
  private String streetAddress;

  /**
   * Constructs a MockPerson with the given information.
   *
   * @param firstName     First Name
   * @param lastName      Last Name
   * @param datetime      Datetime
   * @param emailAddress  Email Address
   * @param gender        Gender
   * @param streetAddress Street Address
   */
  public MockPerson(String firstName, String lastName, String datetime,
                    String emailAddress, String gender, String streetAddress) {
    this.firstName = firstName;
    this.lastName = lastName;
    this.datetime = datetime;
    this.emailAddress = emailAddress;
    this.gender = gender;
    this.streetAddress = streetAddress;
  }

  /**
   * Returns a MockPerson's information in a ready-to-print format.
   *
   * @return String containing MockPerson's information
   */
  public String toString() {
    return "Name: " + firstName + " " + lastName
        + "; Datetime: " + datetime
        + "; Email Address: " + emailAddress
        + "; Gender: " + gender
        + "; Street Address: " + streetAddress;
  }

  /**
   * Given a list of MockPerson, returns their information separated into lines.
   *
   * @param mockPeople A list of MockPerson
   * @return String containing all person info separated into lines
   */
  public static String toString(List<MockPerson> mockPeople) {
    String output = "";
    for (int i = 0; i < mockPeople.size(); i++) {
      output += mockPeople.get(i).toString() + ((i == mockPeople.size() - 1) ? "" : "\n");
    }
    return output;
  }
}
