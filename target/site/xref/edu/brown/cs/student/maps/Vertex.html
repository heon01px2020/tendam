4   import java.util.List;

6   /**
7    * Interface that can be implemented by any object that represents a vertex.
8    * @param <V> any type that implements the vertex interface.
9    * @param <E> any type that implements the edge interface.
10   */
11  public interface Vertex<V extends Vertex<V, E>, E extends Edge<V, E>> {
12    /**
13     * Accesses a vertex's id.
14     * @return id
15     */
16    String getID();
17  
18    /**
19     * Finds the vertex's neighbors.
20     * @return list of paths from the current vertex to its neighbors
21     * @throws SQLException if any errors occur during SQL query
22     */
23    List<E> findNeighbors() throws SQLException;
24  
25    /**
26     * Gets an edge to self with a specified weight.
27     * @param weight Specified weight for the edge to have
28     * @return an edge to self
29     */
30    E edgeToSelf(double weight);
31  
32    /**
33     * Accesses a vertex's coordinate.
34     * @return coordinate
35     */
36    double[] getCoordinate();
37  }
