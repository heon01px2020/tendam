package edu.brown.cs.student.maps;

import edu.brown.cs.student.stars.Command;
import edu.brown.cs.student.stars.KDTree;

import java.sql.ResultSet;
import java.sql.SQLException;
import java.util.*;

/**
 * Class for the map command.
 */
public class MapCommand implements Command {

  private static KDTree<Node> kdTreeNodes;
  private static Database db;
  private static Pathfinder pathfinder = new Pathfinder();

  @Override
  public void execute(String params) throws SQLException {
    if (db != null) { // if there is a previous database
      db.getConnection().close(); //close the previous connection to save memory
      pathfinder.getNodeNeighborCache().invalidateAll(); //clear existing cache in pathfinder
    }
    setKDTreeNodes(null);

    if (loadDatabase(params)) {
      if (checkColumns()) {
        buildKDTree();
        System.out.println("map set to " + params);
      } else {
        resetDB();
        System.err.println("ERROR: missing required tables or columns");
      }
    } else {
      resetDB();
    }
  }

  /**
   * Reset database by setting it to null.
   */
  public static void resetDB() {
    db = null;
  }

  /**
   * Builds KDTree and sets it for instance variable.
   * @param nodes List of nodes to build KDTree using
   */
  public static void setKDTreeNodes(List<Node> nodes) {
    if (nodes == null) {
      kdTreeNodes = null;
    } else {
      kdTreeNodes = new KDTree(2, nodes);
    }
  }

  /**
   * Accesses the database.
   *
   * @return database
   */
  public static Database getDatabase() {
    return db;
  }

  /**
   * Gets Pathfinder object.
   * @return Pathfinder
   */
  public static Pathfinder getPathfinder() {
    return pathfinder;
  }

  /**
   * Loads a database from the given file path.
   *
   * @param filepath path to database
   * @return true if database is successfully loaded
   */
  public static boolean loadDatabase(String filepath) {
    db = new Database(filepath);
    return db.getConnection() != null;
  }

  /**
   * Checks if the database loaded contains the expected columns in the expected tables.
   *
   * @return boolean
   * @throws SQLException if any error occurs during sql query
   */
  public boolean checkColumns() throws SQLException {
    boolean columnsExist;
    List<String> nodeColumns = new ArrayList<>(Arrays.asList("id", "latitude", "longitude"));
    List<String> wayColumns = new ArrayList<>(Arrays.asList("id", "start", "end", "name", "type"));
    columnsExist =
        db.checkColumnsExist(nodeColumns, "Node")
            && db.checkColumnsExist(wayColumns, "Way");
    return columnsExist;
  }

  /**
   * Builds a KD Tree with all the traversable nodes from the database.
   *
   * @throws SQLException if an error occurs in any SQL query.
   */
  public void buildKDTree() throws SQLException {
    ResultSet rs = MapCommand.db.runSQLQuery(
        "SELECT DISTINCT Node.id, Node.latitude, Node.longitude "
            + "from Way JOIN Node ON (Way.start = Node.id OR Way.end = Node.id)"
            + " AND (Way.type NOT IN (\"unclassified\", \"\"));");

    List<Node> nodes = new ArrayList<>();

    while (rs.next()) {
      Node n = new Node(rs.getString(1),
          rs.getDouble("latitude"),
          rs.getDouble("longitude"));
      nodes.add(n);
    }
    //kdTreeNodes = new KDTree(2, nodes);
    setKDTreeNodes(nodes);
  }

  /**
   * Accesses the nodes stored in KDTree.
   *
   * @return nodes
   */
  public static KDTree getNodes() {
    return kdTreeNodes;
  }

}
