package edu.brown.cs.student.maps;

import com.google.common.cache.CacheBuilder;
import com.google.common.cache.CacheLoader;
import com.google.common.cache.LoadingCache;

import java.sql.SQLException;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.PriorityQueue;
import java.util.concurrent.ExecutionException;
import java.util.concurrent.TimeUnit;

/**
 * Class for pathfinding algorithms.
 * @param <V> any type that implements the vertex interface
 * @param <E> any type that implements the edge interface
 */
public class Pathfinder<V extends Vertex<V, E>, E extends Edge<V, E>> {

  /**
   * Finds the shortest edge from start to end using dijkstra's algorithm.
   * @param start start vertex
   * @param end end vertex
   * @return a map that maps a node ID to the previous edge that leads to it
   * @throws ExecutionException if errors occur during caching
   */
  public Map<String, E> dijkstra(V start, V end) throws ExecutionException {
    return findPath(start, end, false);
  }

  /**
   * Finds the shortest edge from start to end using A* algorithm.
   * @param start start vertex
   * @param end end vertex
   * @return a map that maps a node ID to the previous edge that leads to it
   * @throws ExecutionException if errors occur during caching
   */
  public Map<String, E> aStar(V start, V end) throws ExecutionException {
    return findPath(start, end, true);
  }

  /**
   * Finds the shortest edge between two vertices.
   * @param start start vertex
   * @param end end vertex
   * @param weighted whether if the distances are weighted
   * @return a map that maps a node ID to the previous edge that leads to it
   * @throws ExecutionException if errors occur during caching
   */
  public Map<String, E> findPath(V start, V end, boolean weighted) throws ExecutionException {
    V currentNode = start;
    boolean foundPath = false;
    // maps startNode's ID to path
    Map<String, E> solution = new HashMap();
    PriorityQueue<E> visited = new PriorityQueue<>();
    double weight = weighted ? heuristic(start, end) : 0;
    visited.add(currentNode.edgeToSelf(weight)); // initial node
    while (!visited.isEmpty()) {
      E minPath = visited.poll();
      if (minPath.getEnd().getID().equals(end.getID())) {
        solution.put(minPath.getEnd().getID(), minPath);
        foundPath = true;
        break;
      } else if (solution.containsKey(minPath.getEnd().getID())) {
        // goes into the next iteration of the loop
        continue;
      } else {
        if (minPath.getWeightedDistance() != 0) { // if this is not path to a node itself
          solution.put(minPath.getEnd().getID(), minPath);
        }
        currentNode = minPath.getEnd();
        // List<Path> neighbors = currentNode.findNeighbors(); // version without caching
        List<E> neighbors = nodeNeighborCache.get(currentNode);
        for (E n : neighbors) {
          if (weighted) {
            n.setWeightedDistance(n.getWeightedDistance() + heuristic(n.getEnd(), end));
          }
          visited.add(n);
        }
      }
    }
    if (foundPath) {
      return solution;
    } else {
      return null;
    }
  }

  /**
   * Calculates the heuristic.
   * @param currentNode current vertex
   * @param endNode target search vertex
   * @return distance between current vertex to the target vertex
   */
  public double heuristic(V currentNode, V endNode) {
    double[] currentCoors = currentNode.getCoordinate();
    double[] endCoors = endNode.getCoordinate();
    return Haversine.distance(currentCoors[0], currentCoors[1], endCoors[0], endCoors[1]);
  }

  private static final int MAX_CACHE_SIZE = 10000;
  private static final int CACHE_EXPIRE_TIME = 10;

  private LoadingCache<V, List<E>> nodeNeighborCache = CacheBuilder.newBuilder()
      .maximumSize(MAX_CACHE_SIZE)
      .expireAfterWrite(CACHE_EXPIRE_TIME, TimeUnit.MINUTES)
      .build(
          new CacheLoader<>() {
            public List<E> load(V n) throws SQLException {
              return n.findNeighbors();
            }
          });

  /**
   * Gets nodeNeighborCache.
   * @return nodeNeighborCache
   */
  public LoadingCache<V, List<E>> getNodeNeighborCache() {
    return nodeNeighborCache;
  }
}
