12  /**
13   * Class for the ways command.
14   */
15  public class WaysCommand implements Command {
16  
17    @Override
18    public void execute(String params) throws SQLException {
19      if (MapCommand.getDatabase() == null) {
20        System.err.println("ERROR: database not loaded yet");
21        return;
22      }
23      if (Common.checkRegex(params,
24          "^-?\\d+(?:\\.\\d+)?\\s-?\\d+(?:\\.\\d+)?\\s-?\\d+(?:\\.\\d+)?\\s-?\\d+(?:\\.\\d+)?")) {
25        String[] p = params.split(" ");
26        double lat1 = Double.parseDouble(p[0]);
27        double lon1 = Double.parseDouble(p[1]);
28        double lat2 = Double.parseDouble(p[2]);
29        double lon2 = Double.parseDouble(p[3]);
30  
31        if (lat1 < lat2 || lon1 > lon2) {
32          System.err.println("ERROR: (lat1, lon1) is not the northwest point");
33          return;
34        }
35  
36        List<String> waysFound = findWays(lat1, lon1, lat2, lon2);
37        for (String id : waysFound) {
38          System.out.println(id);
39        }
40  
41      } else {
42        System.err.println("ERROR: Wrong parameters");
43      }
44    }
45  
46    /**
47     * Finds all ways that lie inside the bounding box with northwest point (lat1, lon1)
48     * and southeast point (lat2, lon2).
49     * @param lat1 latitude of northwest point
50     * @param lon1 longitude of northwest point
51     * @param lat2 latitude of southeast point
52     * @param lon2 longitude of northwest point
53     * @return a list of waysID sorted in alphabetical order
54     * @throws SQLException if an error occurs in any SQL query.
55     */
56    public List<String> findWays(double lat1, double lon1, double lat2, double lon2)
57        throws SQLException {
58      List<String> waysFound = new ArrayList<>();
59      List<String> vals = new ArrayList<>();
60      vals.add(String.valueOf(lat2));
61      vals.add(String.valueOf(lat1));
62      vals.add(String.valueOf(lon1));
63      vals.add(String.valueOf(lon2));
64      ResultSet rs = MapCommand.getDatabase().runSQLQuery(
65          "SELECT DISTINCT Way.id from Way JOIN Node WHERE "
66              + "((Way.start = Node.id OR Way.end = Node.id) AND Node.latitude BETWEEN "
67              + lat2 + " AND " + lat1 + " AND Node.longitude BETWEEN " + lon1 + " AND " + lon2
68              + ") ORDER BY Way.id;");
69  
70      while (rs.next()) {
71        waysFound.add(rs.getString("id"));
72      }
73  
74      return waysFound;
75    }
