13    @Override
14    public void execute(String params) {
15  
16      if (Common.getAllStars() == null) {
17        System.err.println("ERROR: Star data not loaded yet");
18      } else {
19  
20        if (Common.checkRegex(params,
21            "^\\d+\\s-?\\d+(?:\\.\\d+)?\\s-?\\d+(?:\\.\\d+)?\\s-?\\d+(?:\\.\\d+)?")) {
22  
23          String[] p = params.split(" ");
24          List<Star> stars = findKNearestNeighbor(
25              Integer.parseInt(p[0]),
26              Double.parseDouble(p[1]),
27              Double.parseDouble(p[2]),
28              Double.parseDouble(p[3]));
29  
30          String output = Star.toString(stars);
31          if (!output.equals("")) {
32            System.out.println(Star.toString(stars));
33          }
34  
35        } else if (Common.checkRegex(params, "^\\d+\\s\"([^\"])*\"")) {
36  
37          String[] p = params.split(" ", 2);
38          List<Star> stars = findKNearestNeighbor(Integer.parseInt(p[0]), Common.removeQuotes(p[1]));
39  
40          String output = Star.toString(stars);
41          if (!output.equals("")) {
42            System.out.println(Star.toString(stars));
43          }
44  
45        } else {
46          System.err.println("ERROR: Wrong parameters");
47        }
48      }
49    }

51    /**
52     * Finds the closest k stars to a given location.
53     *
54     * @param k Number of neighbors to be found
55     * @param x x coordinates of the location
56     * @param y y coordinates of the location
57     * @param z z coordinates of the location
58     * @return A list of k closest stars
59     */
60    public List<Star> findKNearestNeighbor(int k, double x, double y, double z) {
61      double[] currentCoordinate = new double[]{x, y, z};
62      Common.setStarDistanceFrom(currentCoordinate, Common.getAllStars());
63      Collections.sort(Common.getAllStars());
64      return getKStars(k, Common.getAllStars());
65    }
66  
67    /**
68     * Finds the closest k stars to a given star.
69     *
70     * @param k    Number of neighbors to be found
71     * @param name Name of a star
72     * @return A list of k closest stars
73     */
74    public List<Star> findKNearestNeighbor(int k, String name) {
75      Star currentStar = Common.findStarByName(name);
76      List<Star> starsCopy = new ArrayList<>();
77      if (currentStar != null) {
78        starsCopy = Common.copyStarListExcept(name);
79        Common.setStarDistanceFrom(currentStar.getCoordinate(), starsCopy);
80        Collections.sort(starsCopy);
81      }
82      return getKStars(k, starsCopy);
83    }

85    /**
86     * Extracts the first k stars from a list of stars sorted according to distance.
87     * Randomly choose one to be returned if there is a tie in distance.
88     *
89     * @param k     Number of stars to be returned
90     * @param stars A list of stars that have been sorted in ascending order of distance
91     * @return First k stars
92     */
93    public List<Star> getKStars(int k, List<Star> stars) {
94      List<Star> output = new ArrayList<>();
95  
96      if (stars.size() != 0) {
97        List<Star> neighbors = new ArrayList<>();
98  
99        int i = 0;
100       Star tempStar;
101       double maxDistanceSoFar = 0;
102 
103       // take out the ones that potentially need to be printed
104       while ((i < stars.size())
105           && ((stars.get(i).getDistance() <= maxDistanceSoFar) || (neighbors.size() < k))) {
106         tempStar = stars.get(i);
107         neighbors.add(tempStar);
108         maxDistanceSoFar = tempStar.getDistance();
109         i++;
110       }
111 
112       // take out the untied stars first
113       while (neighbors.size() != 0) {
114         boolean a = (neighbors.get(0).getDistance() < maxDistanceSoFar);
115         if (neighbors.get(0).getDistance() < maxDistanceSoFar) {
116           output.add(neighbors.get(0));
117           neighbors.remove(neighbors.get(0));
118           k--;
119         } else {
120           break;
121         }
122       }
123 
124       // for the tied ones, shuffle before taking out
125       i = 0;
126       while ((neighbors.size() != 0) && (i < k)) {
127         Collections.shuffle(neighbors);
128         output.add(neighbors.get(0));
129         neighbors.remove(neighbors.get(0));
130         i++;
131       }
132     }
133 
134     return output;
135 
136   }
