3   /**
4    * Class for calculating haversine distance.
5    */
6   public final class Haversine {
7     private static final int RADIUS_EARTH = 6371; // Approximation of Earth radius in km
8   
9     private Haversine() {
10    }
11  
12    /**
13     * Calculates the haversine distance between two coordinates.
14     * @param lat1 latitude of point 1
15     * @param lon1 longitude of point 1
16     * @param lat2 latitude of point 2
17     * @param lon2 longitude of point 2
18     * @return haversine distance
19     */
20    public static double distance(double lat1, double lon1, double lat2, double lon2) {
21  
22      double lat1rad = Math.toRadians(lat1);
23      double lat2rad = Math.toRadians(lat2);
24  
25      double dLat  = Math.toRadians((lat2 - lat1));
26      double dLon = Math.toRadians((lon2 - lon1));
27  
28      // h = haversine of central angle
29      double h = haversine(dLat) + Math.cos(lat1rad) * Math.cos(lat2rad) * haversine(dLon);
30      double d = 2 * RADIUS_EARTH * Math.asin(Math.sqrt(h));
31  
32      return d;
33    }
34  
35    /**
36     * Calculates the haversine based on formula.
37     * @param v a value
38     * @return haversine
39     */
40    public static double haversine(double v) {
41      return Math.pow(Math.sin(v / 2), 2);
42    }
43  }
