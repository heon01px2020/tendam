package edu.brown.cs.student.maps;

import edu.brown.cs.student.stars.*;

import java.sql.SQLException;
import java.util.List;

/**
 * Class for the nearest command.
 */
public class NearestCommand implements Command {
  @Override
  public void execute(String params) throws SQLException {
    if (MapCommand.getDatabase() == null) {
      System.err.println("ERROR: database not loaded yet");
      return;
    }
    if (Common.checkRegex(params,
        "^-?\\d+(?:\\.\\d+)?\\s-?\\d+(?:\\.\\d+)?")) {
      String[] p = params.split(" ");
      Node found = findNearestTraversableNode(Double.parseDouble(p[0]), Double.parseDouble(p[1]));
      if (found != null) {
        System.out.println(found.toString());
      }

    } else {
      System.err.println("ERROR: Wrong parameters");
    }
  }

  /**
   * Finds the nearest traversable node to the input point.
   *
   * @param lat latitude of input point
   * @param lon longitude of input point
   * @return nearest traversable node
   * @throws SQLException if an error occurs in any SQL query
   */
  public Node findNearestTraversableNode(double lat, double lon) throws SQLException {
    KDTree<Node> nodes = MapCommand.getNodes();
    List<Node> foundNode = nodes.findKNN(1, new double[]{lat, lon});
    return (foundNode.isEmpty()) ? null : foundNode.get(0);
  }
}
