package edu.brown.cs.student.stars;

import java.io.FileReader;
import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

/**
 * Class for the mock command.
 */
public class Mock implements Command {

  private final List<String> EXPECTED_HEADERS = new ArrayList<>(
      Arrays.asList("First Name", "Last Name", "Email Address",
          "Gender", "Datetime", "Street Address"));
  private final List<String> EXPECTED_DATA_FORMAT = new ArrayList<>(
      Arrays.asList(
          "(?:.*)?", "(?:.*)?",
          "(?:([a-zA-Z0-9_\\-\\.]+)@([a-zA-Z0-9_\\-\\.]+).([a-zA-Z]{2,5}))?", "(?:.*)?",
          "(?:\\d{2,4}[_\\-\\./-]\\d{1,2}[_\\-\\./-]\\d{1,2})?", "(?:.*)?"));

  @Override
  public void execute(String params) {
    Pattern pattern = Pattern.compile("[^\\s]+");
    Matcher matcher = pattern.matcher(params);
    if (matcher.matches()) {
      printMockPeople(params);
    } else {
      System.err.println("ERROR: Wrong Parameters");
    }

  }

  /**
   * Prints all the mockPerson objects loaded from a CSV file.
   * Raises an error if file is not found, or contains invalid data.
   *
   * @param filePath path to the CSV file containing data about mockPerson
   */
  public void printMockPeople(String filePath) {
    try {
      CSVParser parser = new CSVParser(new FileReader(filePath, StandardCharsets.UTF_8));
      List<LinkedHashMap<String, String>> data = parser.parseAll();
      boolean correctFormat = CSVParser.checkFormat(data, EXPECTED_HEADERS, EXPECTED_DATA_FORMAT);
      if (correctFormat) {
        List<MockPerson> mockPeople = new ArrayList<>();
        for (LinkedHashMap<String, String> person : data) {
          MockPerson m = new MockPerson(
              person.get("First Name"),
              person.get("Last Name"),
              person.get("Datetime"),
              person.get("Email Address"),
              person.get("Gender"),
              person.get("Street Address"));
          mockPeople.add(m);
        }
        System.out.println(MockPerson.toString(mockPeople));
      } else {
        System.err.println("ERROR: File contains invalid data");
      }
    } catch (IOException ioe) {
      System.err.println("ERROR: " + ioe.getMessage());
    }
  }
}
