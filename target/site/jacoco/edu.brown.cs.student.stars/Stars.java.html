package edu.brown.cs.student.stars;

import java.io.FileReader;
import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.HashMap;
import java.util.LinkedHashMap;
import java.util.List;

/**
 * Class for the stars command.
 */

public class Stars implements Command {

  private final List<String> EXPECTED_HEADERS = new ArrayList<>(
      Arrays.asList("StarID", "ProperName", "X", "Y", "Z"));
  private final List<String> EXPECTED_DATA_FORMAT = new ArrayList<>(
      Arrays.asList("^\\d+", "(?:.*)?", "-?\\d+(?:\\.\\d+)?",
          "-?\\d+(?:\\.\\d+)?", "-?\\d+(?:\\.\\d+)?"));


  @Override
  public void execute(String params) {
    if (Common.checkRegex(params, "[^\\s]+")) {
      String str = loadStarData(params);
      if (str.substring(0, 5).equals("ERROR")) {
        System.err.println(str);
      } else {
        System.out.println(str);
      }
    } else {
      System.err.println("ERROR: Wrong parameters");
    }
  }

  /**
   * Loads star data from filepath, stores it in Common.allStars.
   * Reports an error if file is not found, file contains malformed data, or if file is empty
   *
   * @param filePath Path to the file containing star data
   * @return a String confirming the number of stars read
   *         or a String indicating error
   */
  public String loadStarData(String filePath) {
    try {
      CSVParser parser = new CSVParser(new FileReader(filePath, StandardCharsets.UTF_8));
      List<LinkedHashMap<String, String>> starData = parser.parseAll();
      //if (starData != null) {
      boolean correctFormat =
          CSVParser.checkFormat(starData, EXPECTED_HEADERS, EXPECTED_DATA_FORMAT);
      if (correctFormat) {
        List<Star> allStars = new ArrayList<>();
        for (HashMap<String, String> star : starData) {
          Star s = new Star(
              Integer.parseInt((star.get("StarID"))),
              star.get("ProperName"),
              Double.parseDouble(star.get("X")),
              Double.parseDouble(star.get("Y")),
              Double.parseDouble(star.get("Z")));
          allStars.add(s);
        }
        Common.setAllStars(allStars);
        int numStars = Common.getAllStars().size();
        return "Read " + numStars + " stars from " + filePath;
      } else {
        return "ERROR: CSV file contains malformed data";
      }
//      } else {
//        return "ERROR: Empty data file";
//      }
    } catch (IOException ioe) {
      return "ERROR: " + ioe.getMessage();
    }
  }
}
