60     * to the traversable node closest to (lat2, lon2).
61     *
62     * @param lat1   latitude of the northwest point
63     * @param lon1   longitude of the northwest point

65     * @param lon2   longitude of the southeast point
66     * @param option if 0, apply A* algorithm, if 1, apply dijkstra algorithm

72                                          int option)
73        throws SQLException, ExecutionException {

83    /**
84     * Find the shortest path from the intersection of Street 1 and Cross-street 1
85     * to the intersection of Street 2 and Cross-street 2.
86     *
87     * @param street1      street 1
88     * @param crossStreet1 Cross-street 1
89     * @param street2      street 2
90     * @param crossStreet2 Cross-street 2
91     * @param option Option for a* or dijkstra; a* if option is 0, dijkstra if option is 1
92     * @return list of string representing the path
93     * @throws SQLException       if any error occurs during the SQL query
94     * @throws ExecutionException if errors occur during caching
95     */
96    public List<String> shortestPathIntersection(String street1, String crossStreet1,
97                                                 String street2, String crossStreet2, int option)
98        throws SQLException, ExecutionException {
99      if ((street1.equals(crossStreet1)) || (street2.equals(crossStreet2))) {
100       System.err.println("ERROR: cannot find intersection between the same street");
101       return new ArrayList<>();
102     }
103     Node intersect1 = findIntersection(street1, crossStreet1);
104     Node intersect2 = findIntersection(street2, crossStreet2);
105     if ((intersect1 == null) || (intersect2 == null)) {
106       System.err.println("ERROR: streets do not exist, are non-traversable, or do not intersect");
107       return new ArrayList<>();
108     }
109     return shortestPathNodes(intersect1.getCoordinate()[0], intersect1.getCoordinate()[1],
110         intersect2.getCoordinate()[0], intersect2.getCoordinate()[1], option);
111   }

149   /**
150    * Parses the shortest paths into a list of output string.
151    *
152    * @param paths       shortest paths found
153    * @param startNodeID start node
154    * @param endNodeID   end node
155    * @return list of strings, each representing the path from one node to the next
156    */
157   public List<String> parsePaths(Map<String, Path> paths, String startNodeID, String endNodeID) {
158     LinkedList<String> output = new LinkedList<>();
159     if (paths == null) { // if no path is found
160       output.add(startNodeID + " -/- " + endNodeID);
161     } else {
162       String currentNode = endNodeID;
163       while (!currentNode.equals(startNodeID)) {
164         Path p = paths.get(currentNode);
165         String path = p.toString();
166         output.addFirst(path);
167         currentNode = p.getStart().getID();
168       }
169     }
170     return output;
171   }
