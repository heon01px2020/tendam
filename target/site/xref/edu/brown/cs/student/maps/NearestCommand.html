8   /**
9    * Class for the nearest command.
10   */
11  public class NearestCommand implements Command {
12    @Override
13    public void execute(String params) throws SQLException {
14      if (MapCommand.getDatabase() == null) {
15        System.err.println("ERROR: database not loaded yet");
16        return;
17      }
18      if (Common.checkRegex(params,
19          "^-?\\d+(?:\\.\\d+)?\\s-?\\d+(?:\\.\\d+)?")) {
20        String[] p = params.split(" ");
21        Node found = findNearestTraversableNode(Double.parseDouble(p[0]), Double.parseDouble(p[1]));
22        if (found != null) {
23          System.out.println(found.toString());
24        }
25  
26      } else {
27        System.err.println("ERROR: Wrong parameters");
28      }
29    }
30  
31    /**
32     * Finds the nearest traversable node to the input point.
33     *
34     * @param lat latitude of input point
35     * @param lon longitude of input point
36     * @return nearest traversable node
37     * @throws SQLException if an error occurs in any SQL query
38     */
39    public Node findNearestTraversableNode(double lat, double lon) throws SQLException {
40      KDTree<Node> nodes = MapCommand.getNodes();
41      List<Node> foundNode = nodes.findKNN(1, new double[]{lat, lon});
42      return (foundNode.isEmpty()) ? null : foundNode.get(0);
43    }
44  }
