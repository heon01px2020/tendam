package edu.brown.cs.student.maps;

import edu.brown.cs.student.maps.MapCommand;
import edu.brown.cs.student.stars.Command;
import edu.brown.cs.student.stars.Common;

import java.sql.ResultSet;
import java.sql.SQLException;
import java.util.ArrayList;
import java.util.List;

/**
 * Class for the ways command.
 */
public class WaysCommand implements Command {

  @Override
  public void execute(String params) throws SQLException {
    if (MapCommand.getDatabase() == null) {
      System.err.println("ERROR: database not loaded yet");
      return;
    }
    if (Common.checkRegex(params,
        "^-?\\d+(?:\\.\\d+)?\\s-?\\d+(?:\\.\\d+)?\\s-?\\d+(?:\\.\\d+)?\\s-?\\d+(?:\\.\\d+)?")) {
      String[] p = params.split(" ");
      double lat1 = Double.parseDouble(p[0]);
      double lon1 = Double.parseDouble(p[1]);
      double lat2 = Double.parseDouble(p[2]);
      double lon2 = Double.parseDouble(p[3]);

      if (lat1 < lat2 || lon1 > lon2) {
        System.err.println("ERROR: (lat1, lon1) is not the northwest point");
        return;
      }

      List<String> waysFound = findWays(lat1, lon1, lat2, lon2);
      for (String id : waysFound) {
        System.out.println(id);
      }

    } else {
      System.err.println("ERROR: Wrong parameters");
    }
  }

  /**
   * Finds all ways that lie inside the bounding box with northwest point (lat1, lon1)
   * and southeast point (lat2, lon2).
   * @param lat1 latitude of northwest point
   * @param lon1 longitude of northwest point
   * @param lat2 latitude of southeast point
   * @param lon2 longitude of northwest point
   * @return a list of waysID sorted in alphabetical order
   * @throws SQLException if an error occurs in any SQL query.
   */
  public List<String> findWays(double lat1, double lon1, double lat2, double lon2)
      throws SQLException {
    List<String> waysFound = new ArrayList<>();
    List<String> vals = new ArrayList<>();
    vals.add(String.valueOf(lat2));
    vals.add(String.valueOf(lat1));
    vals.add(String.valueOf(lon1));
    vals.add(String.valueOf(lon2));
    ResultSet rs = MapCommand.getDatabase().runSQLQuery(
        "SELECT DISTINCT Way.id from Way JOIN Node WHERE "
            + "((Way.start = Node.id OR Way.end = Node.id) AND Node.latitude BETWEEN "
            + lat2 + " AND " + lat1 + " AND Node.longitude BETWEEN " + lon1 + " AND " + lon2
            + ") ORDER BY Way.id;");

    while (rs.next()) {
      waysFound.add(rs.getString("id"));
    }

    return waysFound;
  }

}
