package edu.brown.cs.student.stars;

import java.util.Arrays;
import java.util.List;
import java.util.Objects;

/**
 * Class that represents a star.
 */
//public class Star<Star extends HasCoordinate<Star>> implements HasCoordinate<Star> {
public class Star implements HasCoordinate, Comparable<Star> {
  private int starID;
  private String name;
  private double[] coordinate;
  private double distance;

  /**
   * Construtor to set up a Star.
   *
   * @param starID star's unique ID
   * @param name   star's name
   * @param x      x coordinate of star's location
   * @param y      y coordinate of star's location
   * @param z      z coordinate of star's location
   */
  public Star(int starID, String name, double x, double y, double z) {
    this.starID = starID;
    this.name = name;
    this.coordinate = new double[]{x, y, z};
  }

  /**
   * Accesses a star's StarID.
   * @return StarID
   */
  public int getID() {
    return starID;
  }

  /**
   * Accesses a star's name.
   * @return Star name
   */
  public String getName() {
    return name;
  }

  /**
   * Accesses a star's coordinate.
   * @return coordinate
   */
  @Override
  public double[] getCoordinate() {
    return Arrays.copyOf(coordinate, coordinate.length);
  }

  /**
   * Accesses a star's distance from a point.
   * @return distance
   */
  public double getDistance() {
    return distance;
  }

  /**
   * Mutates a star's distance.
   * @param d the star's distance from another point
   */
  @Override
  public void setDistance(double d) {
    distance = d;
  }

  /**
   * Overrides hashCode.
   * @return Positive number if greater, 0 if same, negative if less
   */
  @Override
  public int hashCode() {
    return Objects.hash(getSigFields());
  }

  /**
   * Overriding equality for Star object.
   *
   * @param secondStar Star to be checked if equal to this Star.
   * @return True if equal, false if unequal.
   */
  @Override
  public boolean equals(Object secondStar) {
    if (this == secondStar) {
      return true;
    }
    if (!(secondStar instanceof Star)) {
      return false;
    }
    Star second = (Star) secondStar;
    for (int i = 0; i < this.getSigFields().length; ++i) {
      if (!Objects.equals(this.getSigFields()[i], second.getSigFields()[i])) {
        return false;
      }
    }
    return true;
  }

  /**
   * Gets the significant fields as an array of objects.
   * @return significant field of star object
   */
  private Object[] getSigFields() {
    Object[] result = {starID, name, coordinate, distance};
    return result;
  }

  /**
   * Compares two stars according to their distance value.
   * @param s Another star
   * @return 0 if distance is equal, 1 if larger, -1 if smaller
   */
  @Override
  public int compareTo(Star s) {
    return Double.compare(distance, s.getDistance());
  }

  /**
   * Returns a star's starID as a String.
   * @return String containing StarID
   */
  @Override
  public String toString() {
    return Integer.toString(starID);
  }

  /**
   * Given a list of stars, returns their StarID separated into lines.
   * @param stars a list of stars
   * @return String containing all StarID separated into lines
   */
  public static String toString(List stars) {
    String output = "";
    for (int i = 0; i < stars.size(); i++) {
      output += stars.get(i).toString() + ((i == stars.size() - 1) ? "" : "\n");
    }
    return output;
  }
}
