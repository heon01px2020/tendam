3   import java.io.File;
4   import java.sql.*;
5   import java.util.List;
6   
7   /**
8    * Class for a database.
9    */
10  public class Database {
11  
12    private Connection conn = null;
13  
14    /**
15     * Database constructor.
16     * @param filepath file name of SQLite3 database to open.
17     */
18    public Database(String filepath) {
19      if (fileExists(filepath)) {
20        try {
21          // this line loads the driver manager class, and must be
22          // present for everything else to work properly
23          Class.forName("org.sqlite.JDBC");
24          String urlToDB = "jdbc:sqlite:" + filepath;
25          conn = DriverManager.getConnection(urlToDB);
26          // tell the database to enforce foreign keys during operations
27          Statement stat = conn.createStatement();
28          stat.executeUpdate("PRAGMA foreign_keys=ON;");
29        } catch (SQLException e) {
30          System.err.println("ERROR: " + e.getMessage());
31        } catch (ClassNotFoundException e) {
32          System.err.println("ERROR: couldn't establish connection to database");
33        }
34      }
35    }
36  
37    /**
38     * Checks if a file exists.
39     * Throws error if file does not exist.
40     * @param filepath path to a file
41     * @return true if filepath exists and is a file
42     */
43    public static boolean fileExists(String filepath) {
44      File file = new File(filepath);
45      if (!file.exists() || !file.isFile()) {
46        System.err.println("ERROR: file does not exist");
47        return false;
48      }
49      return true;
50    }
51  
52    /**
53     * Gets the connection to a database.
54     * @return connection to database
55     */
56    public Connection getConnection() {
57      return conn;
58    }
59  
60    /**
61     * Check if columns exist in a table in a database.
62     * @param columns a list of column names
63     * @param table table name
64     * @return true if table exists in database and if all the columns exist in the table
65     * @throws SQLException if any error occurs in SQL query
66     */
67    public boolean checkColumnsExist(List<String> columns, String table) throws SQLException {
68      boolean columnsExist = true;
69      DatabaseMetaData dbm = conn.getMetaData();
70      for (String column : columns) {
71        // attempt to get "column" from "table" from database
72        ResultSet rs = dbm.getColumns(null, null, table, column);
73        if (!rs.next()) { // Table or Column do not exist
74          columnsExist = false;
75        }
76      }
77      return columnsExist;
78    }
79  
80    /**
81     * Runs sql commands on the current database.
82     * @param sql SQL command
83     * @return ResultSet from sql command
84     * @throws SQLException if any error occurs during SQL query
85     */
86    public ResultSet runSQLQuery(String sql) throws SQLException {
87      if (conn != null) {
88        PreparedStatement prep = conn.prepareStatement(sql);
89        try {
90          return prep.executeQuery();
91        } catch (SQLException e) {
92          return null;
93        }
94      } else {
95        System.err.println("ERROR: no database connection");
96        return null;
97      }
98    }
99  }
